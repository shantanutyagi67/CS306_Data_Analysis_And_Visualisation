    mean_x = sum(x)/len(x)
    t = [(i-mean_x)**2 for i in x]

def calculate_autocorrelation(x):
    n = len(x)
    variance = x.var()
    x = x-x.mean()
    r = np.correlate(x, x, mode = 'full')[-n:]
    result = r/(variance*(np.arange(n, 0, -1)))
    return result
